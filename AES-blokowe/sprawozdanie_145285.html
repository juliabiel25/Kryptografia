def ecb_encrypt(key, text):
    cipher = AES.new(key, AES.MODE_ECB)
    ciphertext = cipher.encrypt(text.encode('utf-8'))
    return ciphertext

def ecb_decrypt(key, ciphertext):
    cipher = AES.new(key, AES.MODE_ECB)
    text = cipher.decrypt(ciphertext).decode('utf-8')
    return text

# =====
# CBC   
# =====

def cbc_encrypt(key, text):
    cipher = AES.new(key, AES.MODE_CBC)
    text_bytes = text.encode('utf-8')        
    ct = cipher.encrypt(text_bytes)
    iv = cipher.iv
    return iv, ct

def cbc_decrypt(iv, key, ct):
    cipher = AES.new(key, AES.MODE_CBC, iv=iv)
    text = cipher.decrypt(ct).decode('utf-8')
    return text   

# =====
# CTR  
# =====

def ctr_encrypt(key, text):
    text_bytes = text.encode('utf-8')
    cipher = AES.new(key, AES.MODE_CTR)
    ct = cipher.encrypt(text_bytes)
    nonce = cipher.nonce        
    return nonce, ct

def ctr_decrypt(nonce, key, ct):
    cipher = AES.new(key, AES.MODE_CTR, nonce=nonce)
    text = cipher.decrypt(ct).decode('utf-8')
    return text

Wnioski¶

We wszystkich testowanych trybach, czas szyfrowania i deszyfrowania jest wprost proporcjonalny do rozmiaru pliku wejciowego. Porwnujc czasy szyfrowania i deszyfrowania w rznych trybach, najszybciej wypady tryby ECB i CTR o bardzo zblionych do siebie czasach. Tryby CBC i OFB rwnie okazay si dziaa z podobn prdkoci - ok. dwa razy wiksz od wspomnianych poprzednio ECB i CTR. Zdecydowanie najduszym czasem pracy wykazao si szyfrowanie z uyciem trybu CFB.
